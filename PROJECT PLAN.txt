utilizador
	id
	nome
	username (login)
	password

master tem: + (extends utilizador)
	id
	lista de admins (não precisa de ser atributo, quero que todos os masters vejam todas as listas)
	lista de salas (não precisa de ser atributo, quero que todos os masters vejam todas as listas)

diretor(admin) tem: + (extends utilizador)
	id
	departamento
	lista de coordenadores (cada um com o seu curso)
	lista de anos
	

coordenador(user) tem: + (extends utilizador)
	id
	curso
	duração do curso
	lista de UC's

ano tem:
	id
	1º semestre
	2º semestre
	epoca especial inicio
	epoca especial fim

semestre tem:
	id
	começo do semestre
	fim do semestre
	epoca de exames inicio
	epoca de exames fim
	epoca de recurso inicio
	epoca de recurso fim
	lista de utilizadores
	lista de cursos (utilizador -> curso)
	mapa

CU tem:
	id
	nome da UC
	numero de estudantes
	tipo de avaliaçao (continua ou n)
	assiduidade
	numero de avaliações pretendidos
	ano (que ano do curso vai ser lecionado)
	semestre
	lista de avaliações

avaliações tem:
	id
	tipo de avaliaçao (teste, apresentaçao, etc..)
	peso da avaliação
	qual epoca da avaliaçao
	sala com computador (true / false)
	exame será feito no tempo de aula? (true false)
	data e hr do começo da avaliaçao
	data e hr do fim da avaliaçao
	sala onde será feita a avaliaçao
	nota minima
	
mapa tem:
	id
	lista de avaliações de todas as ucs

sala tem:
	id
	numero de sala
	designaçao (auditorio, sala de aula, laboratorio,)
	tipo de material (mesas ou computadores)
	nº de lugares
	edificio (nº de piso ou outro edificio)
